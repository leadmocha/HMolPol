70  // Open file with NIST materials
71  std::ifstream nist_materials;
72  nist_materials.open("nist_materials.txt");
73  // Check whether materials file could be opened correctly
74  if (nist_materials.good())
75  {
76  //add all the NIST materials we want to use
77  std::string material;
78  // Loop over all lines in the file (ie all the needed NIST materials)
79  while (! nist_materials.eof())
80  {
81  nist_materials >> material;
82  // Try to load the NIST material
83  G4bool success = NistManager->FindOrBuildMaterial(material);
84  // And complain if we can't find the material
85  if (! success)
86  {
87  G4cout << "Material " << material << " not found!" << G4endl;
88  }
89  }
90  } else
91  {
92  // If the file cannot be found, load only the vacuum
93  NistManager->FindOrBuildMaterial("G4_Galactic");
94  G4cout << "File nist_materials.txt not found. "
95  << "Only loading NIST vacuum!" << G4endl;
96  }
97  // Close the file
98  nist_materials.close();
99 
100  //if the simulation is using GDML
101  //why don't we need #include <G4GDMLParser.hh>
102  //what's a parser
103  if (fGDMLParser)
104  {
105  //why are you deleting the G4GDMLParser right away
106  delete fGDMLParser;
107  }
108  /// \todo then are you recreating the G4GDMLParser - this makes no sense
109  fGDMLParser = new G4GDMLParser();
110  fGDMLParser->Clear();

115  //read in the geometry - by reading the Mother Volume
116  ///all of the Geometry will be read in since they are all

122  //the world volume is set as the world volume??
123  //but how does this become the MotherVolume??

125 
126  //==========================
127  // List auxiliary info
128  // this includes:
129  // color
130  // alpha (or opaqueness)
131  // detector type - if it is a sensitive detector
132  //==========================
133 
134  // define and auxiliary map which has all the above info in it
135  const G4GDMLAuxMapType* auxmap = fGDMLParser->GetAuxMap();
136 
137  //if it finds auxiliary info tell us

146  for(G4GDMLAuxMapType::const_iterator
147  iter = auxmap->begin();

149  {
150  G4cout << "Volume " << ((*iter).first)->GetName()
151  << " has the following list of auxiliary information: "<< G4endl;
152  for (G4GDMLAuxListType::const_iterator

156  G4cout << "--> Type: " << (*vit).type

158 
159  // Support for the auxiliary tag "Visibility" that can be

163 
164  //if there is a visibility attribute then set it

168  G4Colour colour(1.0,1.0,1.0);

172  //get the new color that is specified by the GDML code

174  colour = visAttribute_old->GetColour();

180  visAttribute_new.SetVisibility(false);
181  if ((*vit).value == "wireframe") // if we want wireframe
182  visAttribute_new.SetForceWireframe(true);
183  // set new visibility attributes
184  ((*iter).first)->SetVisAttributes(visAttribute_new);
185  }
186 
187  // Support for the auxiliary tag "Color" that can be any of black,

190  {
191  G4Colour colour(1.0,1.0,1.0);
192  // get requested color, if it exists
193  if (G4Colour::GetColour((*vit).value, colour))
194  {

196  // create new visibility attributes
197  G4VisAttributes visAttribute(colour);
198  // set new visibility attributes
199  ((*iter).first)->SetVisAttributes(visAttribute);
200  } else
201  {

205  }
206

211  // get the old color - Geant4 standard color
212  G4Colour colour(1.0,1.0,1.0);

216

218  colour = visAttribute_old->GetColour();
